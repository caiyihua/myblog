# 工厂模式

-返回一系列已经预知好的对象 -需要大批量对象的时候运用

function factory(type) {
 // 使得有没有实例化函数 返回的结果一样 （实例化后 其this指向的是factory 否则为window 防错）
 if (this instanceof factory) {
   const res = new this[type]();
   return res;
 } else {
   return new factory(type);
 }
}


factory.prototype = {
 a: function() {
   // ...
 },
 b: function() {
   // ...
 }
}

let A = new factory('a');
let B = factory('a'); // A和B返回的一样
